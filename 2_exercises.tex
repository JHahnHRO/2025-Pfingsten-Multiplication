\begin{problem}[title={Mastertheorem}]\label{ex:mastertheorem}

\[T(n) = a T\left(\frac{n}{b}\right) + f(n)\]

\begin{enumerate}
\item Wenn $f\in O(n^c)$ mit $c<e_{crit}$, dann ist $T\in\Theta(n^{e_{crit}})$.

\end{enumerate}
